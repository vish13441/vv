import React, { useState } from 'react';
import './TestDriveForm.css';

function TestDriveForm() {
  const [name, setName] = useState('');
  const [email, setEmail] = useState('');
  const [phone, setPhone] = useState('');
  const [carModel, setCarModel] = useState('Choose a car model');
  const carModels = ['Model 1', 'Model 2', 'Model 3', 'Model 4'];

  const handleSubmit = (event) => {
    event.preventDefault();
    // handle form submission here
    console.log('Name:', name);
    console.log('Email:', email);
    console.log('Phone:', phone);
    console.log('Car Model:', carModel);
    alert('Thank you for your interest in a test drive! We will be in touch shortly.');
  };

  return (
    <form className="test-drive-form" onSubmit={handleSubmit}>
      <h2>Test Drive Request</h2>
      <div className="form-group">
        <label htmlFor="name">Name:</label>
        <input
          type="text"
          id="name"
          value={name}
          onChange={(event) => setName(event.target.value)}
        />
      </div>
      <div className="form-group">
        <label htmlFor="email">Email:</label>
        <input
          type="email"
          id="email"
          value={email}
          onChange={(event) => setEmail(event.target.value)}
        />
      </div>
      <div className="form-group">
        <label htmlFor="phone">Phone:</label>
        <input
          type="tel"
          id="phone"
          value={phone}
          onChange={(event) => setPhone(event.target.value)}
        />
      </div>
      <div className="form-group">
        <label htmlFor="carModel">Car Model:</label>
        <select id="carModel" value={carModel} onChange={(event) => setCarModel(event.target.value)}>
          <option disabled>Choose a car model</option>
          {carModels.map((model) => (
            <option key={model} value={model}>
              {model}
            </option>
          ))}
        </select>
      </div>
      <button type="submit">Submit</button>
    </form>
  );
}

export default TestDriveForm;
